<script>
    const API_URL = 'http://localhost:3000/api/clothes';

    async function loadClothes() {
        const res = await fetch(API_URL);
        const list = await res.json();
        const body = document.getElementById('clothesTableBody');
        body.innerHTML = '';
        list.forEach(c => {
            const row = `<tr id="row-${c.barcodeId}">
                <td><img src="${c.imagePath || ''}" onerror="this.src='https://via.placeholder.com/40'" style="width:45px;height:45px;border-radius:8px;object-fit:cover;"></td>
                <td class="text-start">
                    <div class="fw-bold">${c.name}</div>
                    <div class="memo-text">${c.memo || ''}</div>
                </td>
                <td><span class="badge-status status-${c.status}">${c.status}</span></td>
                <td class="small">${c.washDate || '기록없음'}</td>
                <td class="small text-secondary">${c.location}</td>
                <td>
                    <div class="d-flex gap-1 justify-content-center">
                        <button class="btn btn-sm btn-outline-primary" onclick="markAsWashed('${c.barcodeId}')" title="세탁완료"><i class="fas fa-soap"></i></button>
                        <select class="form-select form-select-sm" style="width:75px" onchange="updateStatus('${c.barcodeId}', this.value)">
                            <option value="보관" ${c.status==='보관'?'selected':''}>보관</option>
                            <option value="세탁" ${c.status==='세탁'?'selected':''}>세탁</option>
                            <option value="외출" ${c.status==='외출'?'selected':''}>외출</option>
                        </select>
                        <button class="btn btn-sm btn-outline-danger" onclick="deleteClothes('${c.barcodeId}')"><i class="fas fa-trash"></i></button>
                    </div>
                </td>
            </tr>`;
            body.innerHTML += row;
        });
    }

    async function findAndHighlight(bid) {
        document.querySelectorAll('tr').forEach(t => t.style.backgroundColor = '');
        document.querySelectorAll('.closet-btn').forEach(b => b.style.boxShadow = '');
        const res = await fetch(API_URL);
        const list = await res.json();
        const target = list.find(c => c.barcodeId === bid);
        if (target) {
            const r = document.getElementById(`row-${bid}`);
            if (r) { r.style.backgroundColor = '#fef9c3'; r.scrollIntoView({ behavior: 'smooth', block: 'center' }); }
            document.querySelectorAll('.closet-btn').forEach(b => {
                if (b.innerText === target.location) b.style.boxShadow = '0 0 15px #4f46e5';
            });
            return true;
        }
        return false;
    }

    document.getElementById('clothesForm').addEventListener('submit', async (e) => {
        e.preventDefault();
        const bid = document.getElementById('barcode').value;
        if (await findAndHighlight(bid)) return;

        const formData = new FormData();
        formData.append('barcodeId', bid);
        formData.append('name', document.getElementById('clothesName').value);
        formData.append('category', document.getElementById('category').value);
        formData.append('thickness', document.getElementById('thickness').value);
        formData.append('status', '보관');
        formData.append('memo', document.getElementById('memo').value);
        
        const loc = document.querySelector('input[name="location"]:checked');
        if (!loc) { alert("위치를 고르세요"); return; }
        formData.append('location', loc.value);

        const img = document.getElementById('imageFile').files[0];
        if (img) formData.append('image', img);

        await fetch(API_URL, { method: 'POST', body: formData });
        loadClothes();
        e.target.reset();
    });

    async function markAsWashed(id) { await fetch(`${API_URL}/${id}/wash`, {method:'PUT'}); loadClothes(); }
    async function deleteClothes(id) { if (confirm('삭제할까요?')) { await fetch(`${API_URL}/${id}`, {method:'DELETE'}); loadClothes(); } }
    async function updateStatus(id, s) { await fetch(`${API_URL}/${id}/status?status=${s}`, {method:'PUT'}); loadClothes(); }

    loadClothes();
</script>
